    new feds.Responsifier()
    .add({
        observer: "#example-modal",
        observable: "#modalTrigger",
        event: feds.Responsifier.CLICK,
        toggle: ["display-flex"]
    }).add({
        observer: "#example-modal",
        observable: "#modalClose",
        event: feds.Responsifier.CLICK,
        toggle: ["display-flex"]
    }).add({
        observer: "body",
        observable: "#modalClose",
        event: feds.Responsifier.CLICK,
        toggle: ["scroll-none"]
    }).add({
        observer: "body",
        observable: "#modalTrigger",
        event: feds.Responsifier.CLICK,
        toggle: ["scroll-none"]
    })
    .init();
