Responsifier

The Responsifier component is an implementation of Container Queries. Here is an example of a UI implementation with a responsive menu and body.

    new feds.Responsifier()
    .add({
      observer: "#cqBody",
      observable: window,
      event: feds.Responsifier.EVENT.RESIZE,
      range: [0, 799],
      add: ["flex-dir-col"]
    })
    .add({
      observer: "#cqBody",
      observable: window,
      event: feds.Responsifier.EVENT.RESIZE,
      range: [800, "*"],
      remove: ["flex-dir-col"]
    })
    .add({
      observer: "#cqMainMenu",
      observable: window,
      event: feds.Responsifier.EVENT.RESIZE,
      range: [0, 799],
      add: ["hide"]
    })
    .add({
      observer: "#cqMainMenu",
      observable: window,
      event: feds.Responsifier.EVENT.RESIZE,
      range: [800, "*"],
      remove: ["hide"]
    })
    .add({
      observer: "#cqMainMenu",
      observable: "#cqToggleMenu",
      event: feds.Responsifier.EVENT.CLICK,
      toggle: ["hide"]
    })
    .add({
      observer: "#cqToggleMenu",
      observable: window,
      event: feds.Responsifier.EVENT.RESIZE,
      range: [0, 799],
      remove: ["hide"]
    })
    .add({
      observer: "#cqToggleMenu",
      observable: window,
      event: feds.Responsifier.EVENT.RESIZE,
      range: [800, "*"],
      add: ["hide"]
    })
    .add({
      observer: "#cqContent .column:nth-child(-n+2)",
      observable: window,
      event: feds.Responsifier.EVENT.RESIZE,
      range: [0, 799],
      add: ["half"]
    })
    .add({
      observer: "#cqContent .column:nth-child(-n+2)",
      observable: window,
      event: feds.Responsifier.EVENT.RESIZE,
      range: [800, "*"],
      remove: ["half"]
    })
    .init();
